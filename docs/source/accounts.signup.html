"use strict";
module.exports = [
    "Apiv1Service",
/**
 * @memberof accounts
 * @ngdoc service 
 * @name signup
 * @param {service} Apiv1service
 * @description
 *      Service to signup 
 */
function (Apiv1Service) {
    const signup = function (data) {
        debugger;
        data = Apiv1Service.normalize(data);
        var api = Apiv1Service.getInstance();
        return api.post('accounts/signup', data)
            .then((result) => {
                console.log(result);
            });
    };
    return {
        signup
    };
}];
